pub fn tai2tt_dt(datetime: &NaiveDateTime) -> NaiveDateTime
Expand description

Convert datetime from TAI to TT.

Arguments

  • datetime - Datetime in TAI.

Returns

Returns the datetime in TT.

Returns Error if it fail to convert.

Examples

use convdate;
use chrono::NaiveDate;

let tai = convdate::tai2tt_dt(
    &NaiveDate::from_ymd(2017, 1, 1).and_hms(12, 0, 0));

assert_eq!(tai, NaiveDate::from_ymd(2017, 1, 1).and_hms_milli(12, 0, 32, 184));
